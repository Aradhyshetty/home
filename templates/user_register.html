function validatePassword() { var p = document.getElementById('pas').value, errors = []; if (p.length < 8) { errors.push("Your password must be at least 8 characters"); } if (p.search(/[a-z]/i) < 0) { errors.push("Your password must contain lowercase letters."); } if (p.search(/[A-Z]/) < 0) { errors.push("Your password must contain uppercase letters."); } if (p.search(/[0-9]/) < 0) { errors.push("Your password must contain at least one digit."); } if (p.search(/[!@#$%^&*]/) < 0) { errors.push("Your password must contain special characters."); } if (errors.length > 0) { alert(errors.join("\n")); return false; } return true; }
